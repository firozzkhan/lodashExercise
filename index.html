Array Methods

Exercise 1: _.flatten()

                
                    Flattens array a single level deep.
                    example:
                    var result = _.flatten([1, 2, [4, 3, [6, 7], 3, 2], 9, 8]);

Exercise 2: _.head()

                
                    Gets first Element.
                    example:
                    var result = _.head([1,2,[4,3,[6,7],3,2],9,8]);

Exercise 3: _.indexOf()

                
                    Gets index of First Occurence of Element.
                    example:
                    var result = _.indexOf(_.flatten([1, 2, [4, 3, [6, 7], 3, 2], 9, 8]), 4);

Exercise 4: _.initial()

                
                    Give array excluding last element.   
                    example:
                    var result = _.initial([1,2,[4,3,[6,7],3,2],9,8]);

Exercise 5: _.last()

                
                    Give last element of array.
                    example:
                    var result = _.last([1, 2, [4, 3, [6, 7], 3, 2], 9, 8]);

Exercise 6: _.nth()

                
                    Give element at nth index.
                    If n is -ve then nth element from end is returned.
                    example:
                    var result = _.nth([1, 4, 6, 8, 4, 9], 3);

Exercise 7: _.pull()

                
                    Returns array by removing  given values.
                    example:
                    var result = _.pull([1, 2, 4, 3, 6, 7, 3, 2, 9, 8], 6);

Exercise 8: _.pullAll()

                
                    Returns array by removing  given values in array.
                    example:
                    var result = _.pullAll([1, 2, 4, 3, 6, 7, 3, 2, 9, 8], [3, 6, 7]);

Exercise 9: _.remove

                
                    Gives array of removed values if function returns true.
                    example:  
                    var array = [3, 2, 5, 6, 8, 9]
                    var result = _.remove(array, function (n) {
                        return n % 2 == 0;
                    });

Exercise 10: _.tail

                
                    Give array excluding first element.
                    example:
                    var result = _.tail([1, 2, 4, 3, 6, 7, 3, 2, 9, 8]);

Exercise 11: _.union()

                
                    Give array of unique values of all arrays.
                    example:
                    var result = _.union([2, 4], [3, 4, 6, 7]);

Exercise 12: _.concat()

                
                    Give array by concatenating additional values or array.
                    example:
                    var result = _.concat([2,4], [5,7]);

Exercise 13: _.without()

                
                    Give array excluding given values.
                    example:
                    var result = _.without([2, 3, 4, 3, 2], 2);
